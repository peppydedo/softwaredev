export async function getUser(
  id: string
): Promise<User> {
  const cached = await cache
    .get<User>(`user:${id}`);

  if (cached) return cached;

  const user = await db.users
    .findFirst({ where: { id } });

  await cache.set(
    `user:${id}`, user, 3_600
  );

  return user;
}
